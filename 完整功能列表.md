## 💡 使用技巧

### 技巧1：快速查找企业
```javascript
// 查找海淀区的所有企业
enterpriseData.list.filter(e => e.district === '海淀区');

// 查找人工智能行业的企业
enterpriseData.list.filter(e => e.type === '人工智能');

// 查找高成长性企业
enterpriseData.list.filter(e => e.growth === 'high');
```

### 技巧2：快速查找政策
```javascript
// 查找财政补贴类政策
policyData.list.filter(p => p.type === '财政补贴');

// 查找适用于集成电路的政策
policyData.list.filter(p => p.industry.includes('集成电路'));

// 查找市级政策
policyData.list.filter(p => p.level === '市级');
```

### 技巧3：数据统计分析
```javascript
// 统计各区企业数量
const districtCount = {};
enterpriseData.list.forEach(e => {
    districtCount[e.district] = (districtCount[e.district] || 0) + 1;
});
console.table(districtCount);

// 统计各产业企业数量
const industryCount = {};
enterpriseData.list.forEach(e => {
    industryCount[e.type] = (industryCount[e.type] || 0) + 1;
});
console.table(industryCount);

// 计算平均营收
const avgRevenue = enterpriseData.list.reduce((sum, e) => sum + e.revenue, 0) / enterpriseData.list.length;
console.log('平均营收:', avgRevenue.toFixed(0), '万元');
```
